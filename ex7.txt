from PyPDF2 import PdfReader, PdfWriter

reader = PdfReader("documento_grande.pdf")
writer = PdfWriter()

# Dividir em partes de 50 p√°ginas
for i in range(0, len(reader.pages), 50):
    writer.add_pages(reader.pages[i:i + 50])
    with open(f"parte_{i // 50 + 1}.pdf", "wb") as output_pdf:
        writer.write(output_pdf)
